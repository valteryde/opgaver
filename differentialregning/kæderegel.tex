\begin{question}{multi}
    Bestem $f'(x)$ når $f(x)=(3x + 1)^4$
    \answer{$12(3x+1)^3$}
    \alt{$(3x+1)^3$}
    \alt{$4(3x+1)^3$}
    \alt{$(3x+1)^4$}
    \end{question}
    
    \begin{question}{multi}
    Bestem $f'(x)$ når $f(x)=\ln(5x + 2)$
    \answer{$\frac{5}{5x+2}$}
    \alt{$\frac{1}{5x+2}$}
    \alt{$5x + 2$}
    \alt{$\frac{5}{x}$}
    \end{question}
    
    \begin{question}{multi}
    Bestem $f'(x)$ når $f(x)=e^{2x - 1}$
    \answer{$2e^{2x-1}$}
    \alt{$e^{2x-1}$}
    \alt{$2x \cdot e^{2x-1}$}
    \alt{$2xe^{2x}$}
    \end{question}
    
    \begin{question}{multi}

    \alt{$-\sin(4x-3)$}
