\begin{question}{multi}
    
Løs $f'(x)=0$ når $f(x)=x^2+x$

\answer{$-\frac{1}{2}$}
\alt{$-2$}
\alt{$-\frac{1}{4}$}
\alt{$\frac{1}{2}$}

\end{question}
    
    \begin{question}{multi}
    Løs $f'(x)=0$ når $f(x)=x^3 - 3x$
    \answer{$1$ og $-1$}
    \alt{$-3$}
    \alt{$\frac{1}{2}$}
    \alt{$3$ og $1$}
    \end{question}
    
    \begin{question}{multi}
    Løs $f'(x)=0$ når $f(x)=x^4 - 4x^2$ med hjælpemidler
    \answer{$0$, $\sqrt{2}$ og $-\sqrt{2}$}
    \alt{$0$, $1$ og $-\sqrt{2}$}
    \alt{$0$, $3$ og $-\sqrt{3}$}
    \alt{$0$, $-2$ og $-1$}
    \end{question}
